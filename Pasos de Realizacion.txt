Empezamos empezamos con las clases en los modelos:
    Aquí creamos la clase, con su constructor y solo sus propiedades, y lo exportamos
    Ejemplo:
        class Empresa {
            constructor(nombre, departamentos = []){
                this.nombre = nombre
                this. departamentos = departamentos
            }
        }

        module.exports = Empresa;

Luego continuamos con funciones en los controllers:
    Aquí creamos las funciones que antes existían en las clases:
    Ejemplo: 
        const Empresa = require("../model/Empresa") // Importamos la clase creada anteriormente
        const ServiceEmpresa = require("../services/ServicesEmpresa") // Importamos al que nos ayuda a guardar la información

        exports.agregardepartamentos = async (req, res) => {
            const {nombre, departamentos} = req.body    // Información que nosotros enviamos a traves del PostMan
            const nuevaEmpresa = new Empresa(nombre, departamentos)     // Creamos la instancia 
            const crearEmpresaDepart = await ServiceEmpresa.postEmpresa(nuevaEmpresa.nombre, nuevaEmpresa.departamentos) 
            if (crearEmpresaDepart){
                res.status(201).json({message: "La Empresa y sus departamentos se crearon efectivamente", data: crearEmpresaDepart})
            } else {
                res.status(500).json({message: "Fallo al crear la Empresa y sus departamentos"})
            }
        }

Para tercer paso, tenemos la ruta:
    En ella comenzamos a dictar los métodos del CRUD según la ruta:
    Ejemplo: 
        const express = require('express');
        const router = express.Router();

        const EmpresaController = require("../controllers/EmpresaController") // Importamos el controlador

        router.post("/crearEmpresa", EmpresaController.agregardepartamentos)  // Ejecutamos el método a utilizar (POST, GET, PUT, DELETE)

        module.exports = router;

Por útlimo paso tenemos que agregar la ruta al archivo app.js
    const express = require("express");
